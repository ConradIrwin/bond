def load_yard_gems(*gems)
  @options = gems[-1].is_a?(Hash) ? gems.pop : {}
  require 'yard'
  raise LoadError unless YARD::VERSION >= '0.5.2'
  gems.select {|e| load_yard_gem(e) }
rescue LoadError
  $stderr.puts "Bond Error: yard gem (version >= 0.5.2) not installed "
end
